====== ОБНОВЛЕНИЕ NGINX ОДНОЙ КОМАНДОЙ (БЕЗ SCP) ======

Подключитесь к серверу и выполните:

ssh root@your-server

Затем скопируйте и вставьте ВСЮ эту команду целиком:

cat > /etc/nginx/sites-available/1мост.рф.conf << 'MOST_CONF'
server {
    listen 80;
    listen [::]:80;
    server_name 1мост.рф www.1мост.рф;

    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name 1мост.рф www.1мост.рф;

    ssl_certificate /etc/letsencrypt/live/1мост.рф/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/1мост.рф/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    root /var/www/prohelper_marketing/client;
    index index.html;

    location = /dashboard {
        return 301 https://lk.1мост.рф/dashboard;
    }

    location ^~ /dashboard/ {
        return 301 https://lk.1мост.рф$request_uri;
    }

    location = /login {
        return 301 https://lk.1мост.рф/login;
    }

    location = /register {
        return 301 https://lk.1мост.рф/register;
    }

    location = /forgot-password {
        return 301 https://lk.1мост.рф/forgot-password;
    }

    location / {
        try_files $uri $uri.html @ssr;
    }

    location @ssr {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    access_log /var/log/nginx/prohelper_access.log;
    error_log /var/log/nginx/prohelper_error.log;
}
MOST_CONF

cat > /etc/nginx/sites-available/lk.1мост.рф.conf << 'LK_CONF'
server {

LK_CONF

cat > /etc/nginx/sites-available/holdings.1мост.рф.conf << 'HOLDINGS_CONF'
server {
    listen 80;
    listen [::]:80;
    server_name *.1мост.рф;

    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name ~^(?<holding_slug>[^.]+)\.prohelper\.pro$;

    ssl_certificate /etc/letsencrypt/live/1мост.рф-0001/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/1мост.рф-0001/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    root /var/www/prohelper_lk;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    access_log /var/log/nginx/holding_access.log;

HOLDINGS_CONF

sudo ln -sf /etc/nginx/sites-available/1мост.рф.conf /etc/nginx/sites-enabled/
sudo ln -sf /etc/nginx/sites-available/lk.1мост.рф.conf /etc/nginx/sites-enabled/
sudo ln -sf /etc/nginx/sites-available/holdings.1мост.рф.conf /etc/nginx/sites-enabled/

sudo nginx -t && sudo systemctl reload nginx && echo "✅ Nginx успешно обновлен!"
