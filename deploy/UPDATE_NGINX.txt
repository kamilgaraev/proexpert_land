====== КОМАНДЫ ДЛЯ ОБНОВЛЕНИЯ NGINX (КОПИРУЙ И ВСТАВЛЯЙ) ======

=== ШАГ 1: С ЛОКАЛЬНОЙ МАШИНЫ ===

scp deploy/nginx/1мост.рф.conf root@your-server:/tmp/
scp deploy/nginx/lk.1мост.рф.conf root@your-server:/tmp/
scp deploy/nginx/holdings.1мост.рф.conf root@your-server:/tmp/

=== ШАГ 2: НА СЕРВЕРЕ (одной командой) ===

ssh root@your-server "sudo cp /tmp/1мост.рф.conf /etc/nginx/sites-available/ && sudo cp /tmp/lk.1мост.рф.conf /etc/nginx/sites-available/ && sudo cp /tmp/holdings.1мост.рф.conf /etc/nginx/sites-available/ && sudo ln -sf /etc/nginx/sites-available/1мост.рф.conf /etc/nginx/sites-enabled/ && sudo ln -sf /etc/nginx/sites-available/lk.1мост.рф.conf /etc/nginx/sites-enabled/ && sudo ln -sf /etc/nginx/sites-available/holdings.1мост.рф.conf /etc/nginx/sites-enabled/ && sudo nginx -t && sudo systemctl reload nginx && echo 'Готово!'"

=== ИЛИ ШАГ 2 АЛЬТЕРНАТИВНЫЙ: Подключиться и выполнить по шагам ===

ssh root@your-server

sudo cp /tmp/1мост.рф.conf /etc/nginx/sites-available/
sudo cp /tmp/lk.1мост.рф.conf /etc/nginx/sites-available/
sudo cp /tmp/holdings.1мост.рф.conf /etc/nginx/sites-available/

sudo ln -sf /etc/nginx/sites-available/1мост.рф.conf /etc/nginx/sites-enabled/
sudo ln -sf /etc/nginx/sites-available/lk.1мост.рф.conf /etc/nginx/sites-enabled/
sudo ln -sf /etc/nginx/sites-available/holdings.1мост.рф.conf /etc/nginx/sites-enabled/

sudo nginx -t
sudo systemctl reload nginx

=== ПРОВЕРКА ===

curl -I https://1мост.рф/
curl -I https://1мост.рф/dashboard
curl -I https://lk.1мост.рф/

Ожидаемый результат для /dashboard: HTTP/1.1 301 Moved Permanently
